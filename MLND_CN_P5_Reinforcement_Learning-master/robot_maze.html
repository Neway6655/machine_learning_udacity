import random
import numpy as np

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    if random.random() <= epsilon: # 
        action = np.random.choice(actions, 1)[0]
    else: 
        action = max(qline, key=qline.get)
    return action
